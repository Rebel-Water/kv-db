# Function to simplify adding tests
function(add_gtest test_name)
  add_executable(${test_name} ${ARGN})
  target_link_libraries(${test_name} GTest::gtest_main ZLIB::ZLIB)
  gtest_discover_tests(${test_name})
endfunction()
